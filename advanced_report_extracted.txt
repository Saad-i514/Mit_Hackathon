CATEGORY A — Intelligence Layer
Purpose:  These features replace the original single-source Semantic Scholar search with a parallel 5-database literature engine, deep reagent intelligence from PubChem, protocol similarity search, and clinical trial monitoring. All APIs are completely free with no rate-limit issues for hackathon-scale traffic.
Feature F-01  Multi-Source Literature Engine
Replaces the single Semantic Scholar call with a parallel fan-out across 5 databases. Results are deduplicated by DOI and ranked by a composite score (citation count, recency, abstract relevance). This dramatically improves the QC node's recall — a paper missed by one source is almost always caught by another.

# app/graph/nodes/multi_literature_qc.py
import asyncio, httpx
from app.graph.state import ExperimentState
async def fetch_pubmed(query: str) -> list[dict]:
    base = "https://eutils.ncbi.nlm.nih.gov/entrez/eutils"
    async with httpx.AsyncClient(timeout=12) as c:
        search = await c.get(f"{base}/esearch.fcgi",
            params={"db":"pubmed","term":query,"retmax":5,"retmode":"json","usehistory":"y"})
        ids = search.json()["esearchresult"]["idlist"]
        if not ids: return []
        fetch  = await c.get(f"{base}/efetch.fcgi",
            params={"db":"pubmed","id":",".join(ids),"rettype":"abstract","retmode":"xml"})
    return parse_pubmed_xml(fetch.text)
async def fetch_openalex(query: str) -> list[dict]:
    url = "https://api.openalex.org/works"
    async with httpx.AsyncClient(timeout=12) as c:
        r = await c.get(url, params={"search": query, "per-page": 5,
            "filter": "is_oa:true", "sort": "cited_by_count:desc",
            "mailto": "ai-scientist@yourdomain.com"})
    return [parse_openalex_work(w) for w in r.json().get('results',[])]
async def fetch_biorxiv(query: str) -> list[dict]:
    url = "https://api.biorxiv.org/details/biorxiv/2024-01-01/2025-12-31/0/json"
    async with httpx.AsyncClient(timeout=12) as c:
        r = await c.get(url)
    all_papers = r.json().get('collection',[])
    q_words = set(query.lower().split())
    return [p for p in all_papers if any(w in p.get('title','').lower() for w in q_words)][:5]
async def multi_literature_qc_node(state: ExperimentState) -> ExperimentState:
    hypothesis = state['hypothesis']
    # Run all searches in parallel
    results = await asyncio.gather(
        fetch_pubmed(hypothesis[:200]),
        fetch_openalex(hypothesis[:200]),
        fetch_biorxiv(hypothesis[:150]),
        fetch_semantic_scholar(hypothesis[:200]),   # existing function
        return_exceptions=True
    )
    # Flatten and deduplicate by DOI
    papers = deduplicate_by_doi([p for batch in results
                                 if isinstance(batch, list) for p in batch])
    # Rank by composite score: citations * 0.6 + recency * 0.4
    ranked = sorted(papers, key=composite_score, reverse=True)[:8]
    # Let GPT-4o classify novelty as before
    from app.services.openai_client import classify_novelty
    novelty, refs = await classify_novelty(hypothesis, ranked)
    return {**state, 'novelty_signal': novelty, 'qc_references': refs}

# app/services/pubchem.py
import httpx
from functools import lru_cache
PUBCHEM = 'https://pubchem.ncbi.nlm.nih.gov/rest/pug'
async def enrich_reagent(name: str) -> dict:
    """Fetch full PubChem data for a reagent name."""
    async with httpx.AsyncClient(timeout=10) as c:
        # Step 1: resolve name to CID
        r = await c.get(f'{PUBCHEM}/compound/name/{name}/cids/JSON')
        if r.status_code != 200: return {'name': name, 'pubchem_found': False}
        cid = r.json()['IdentifierList']['CID'][0]
        # Step 2: fetch properties in one call
        props = ['MolecularWeight','MolecularFormula','IUPACName','IsomericSMILES']
        pr = await c.get(f'{PUBCHEM}/compound/cid/{cid}/property/{",".join(props)}/JSON')
        p = pr.json()['PropertyTable']['Properties'][0]
        # Step 3: GHS hazard codes
        hr = await c.get(f'{PUBCHEM}/compound/cid/{cid}/classification/JSON')
        ghs = extract_ghs_codes(hr.json())
        # Step 4: CAS number
        casr = await c.get(f'{PUBCHEM}/compound/cid/{cid}/xrefs/RN/JSON')
        cas = casr.json().get('InformationList',{}).get('Information',[{}])[0].get('RN',[])
    return {
        'cid': cid, 'name': name, 'pubchem_found': True,
        'molecular_weight': p.get('MolecularWeight'),
        'molecular_formula': p.get('MolecularFormula'),
        'cas_number': cas[0] if cas else None,
        'ghs_codes': ghs,
        'pubchem_url': f'https://pubchem.ncbi.nlm.nih.gov/compound/{cid}',
    }

import httpx; from app.config import settings
async def search_protocols(query: str) -> list[dict]:
    async with httpx.AsyncClient(timeout=12) as c:
        r = await c.get('https://www.protocols.io/api/v4/protocols',
            params={'filter':'public','key':query[:150],'page_size':5,
                    'order_field':'citation_count','order_dir':'desc'},
            headers={'Authorization': f'Bearer {settings.PROTOCOLS_IO_TOKEN}'})
    if r.status_code != 200: return []
    items = r.json().get('items', [])
    return [{'title':p['title'],'doi':p.get('doi'),'url':p.get('uri'),
             'citations':p['stats']['number_of_citations'],
             'steps':p['stats']['number_of_steps'],
             'published':p.get('published_on')} for p in items]

Feature F-04  Clinical Trials Radar
Checks whether any active, recruiting, or recently completed clinical trials are already testing a similar hypothesis. Prevents a scientist from investing months in a bench experiment that duplicates an ongoing human study — a real and costly mistake in translational research.
API Reference
Base URL:  https://clinicaltrials.gov/api/v2/studies
Key Required:  No — completely open
Key Parameters:  query.term · filter.overallStatus · pageSize · fields
# app/services/clinical_trials.py
import httpx
CT_BASE = 'https://clinicaltrials.gov/api/v2/studies'
ACTIVE_STATUSES = 'RECRUITING,ACTIVE_NOT_RECRUITING,NOT_YET_RECRUITING,COMPLETED'
async def check_clinical_trials(hypothesis: str) -> dict:
    keywords = extract_key_terms(hypothesis)  # GPT-4o extracts 3-5 key terms
    async with httpx.AsyncClient(timeout=10) as c:
        r = await c.get(CT_BASE, params={
            'query.term': ' AND '.join(keywords),
            'filter.overallStatus': ACTIVE_STATUSES,
            'pageSize': 5,
            'fields': 'NCTId,BriefTitle,OverallStatus,Phase,StartDate,Condition'})
    studies = r.json().get('studies', [])
    return {
        'total_found': r.json().get('totalCount', 0),
        'studies': [{'nct_id': s['protocolSection']['identificationModule']['nctId'],
                     'title': s['protocolSection']['identificationModule']['briefTitle'],
                     'status': s['protocolSection']['statusModule']['overallStatus'],
                     'url': f'https://clinicaltrials.gov/study/{nct_id}'}
                    for s in studies[:3]]}

// Power calculator — pure TypeScript, no external API
function calculateSampleSize(effectSize: number, alpha: number, power: number): number {
  // Two-sample t-test (two-tailed) using iterative normal CDF approximation
  const z_alpha = normalQuantile(1 - alpha / 2)  // e.g. 1.96 for alpha=0.05
  const z_beta  = normalQuantile(power)           // e.g. 0.84 for power=0.80
  const n = 2 * Math.pow((z_alpha + z_beta) / effectSize, 2)
  return Math.ceil(n)  // per group
}

// e.g. for cell viability assays: d=0.8 (large), alpha=0.05, power=0.80

Feature F-09  Interactive Protocol Flowchart
The Protocol tab renders the step-by-step procedure as an interactive directed graph using React Flow. Each node represents one protocol step; edges show the flow. Clicking a node opens a detail panel with full description, duration, materials used, and safety notes. Decision points (e.g. 'If cell viability < 70%, repeat step') are rendered as diamond nodes with branching edges.

// components/plan-viewer/protocol-flowchart.tsx
import { ReactFlow, Background, Controls, MiniMap } from '@xyflow/react'
import '@xyflow/react/dist/style.css'
function protocolToNodes(protocol: ProtocolStep[]) {
  return protocol.map((step, i) => ({
    id: String(i),
    type: step.is_decision ? 'diamond' : 'step',
    position: { x: 300, y: i * 140 },
    data: {
      label: step.title,
      duration: step.duration_hours,
      materials: step.materials_used,
      safety: step.safety_notes,
    }}))
}
function protocolToEdges(protocol: ProtocolStep[]) {
  return protocol.slice(0,-1).map((_,i) => ({
    id: `e${i}-${i+1}`, source: String(i), target: String(i+1),
    animated: true, label: protocol[i].expected_output ?? ''
  }))
}
export function ProtocolFlowchart({ protocol }: { protocol: ProtocolStep[] }) {
  const [selected, setSelected] = useState<ProtocolStep | null>(null)
  return (
    <div style={{ height: 600 }}>
      <ReactFlow nodes={protocolToNodes(protocol)} edges={protocolToEdges(protocol)}
        nodeTypes={{ step: StepNode, diamond: DecisionNode }}
        onNodeClick={(_, node) => setSelected(node.data)}
        fitView>
        <Background /> <Controls /> <MiniMap />
      </ReactFlow>
      {selected && <StepDetailPanel step={selected} onClose={() => setSelected(null)} />}
    </div>
  )
}

// components/plan-viewer/gantt-timeline.tsx
import Gantt from 'frappe-gantt'
import 'frappe-gantt/dist/frappe-gantt.css'
function timelineToGanttTasks(timeline: TimelinePhase[]): GanttTask[] {
  return timeline.map(phase => ({
    id: phase.phase.replace(/\s+/g, '_'),
    name: phase.phase,
    start: weekToDate(phase.week_start),   // converts 'Week 1' -> ISO date
    end:   weekToDate(phase.week_end),
    progress: 0,
    dependencies: phase.depends_on ?? ''
  }))
}
export function GanttTimeline({ timeline }: { timeline: TimelinePhase[] }) {
  const ref = useRef<HTMLDivElement>(null)
  useEffect(() => {
    if (!ref.current) return
    const g = new Gantt(ref.current, timelineToGanttTasks(timeline), {
      view_mode: 'Week',
      on_click: task => showTaskDetail(task),
      on_date_change: (task, start, end) => updateTimeline(task.id, start, end),
    })
    return () => g.destroy()
  }, [timeline])
  return <div ref={ref} className='gantt-container' />
}

  const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' })
  saveAs(blob, 'materials_list.csv')
}
export function exportPDFReport(experiment: ExperimentFull) {
  const pdf = new jsPDF()
  pdf.setFontSize(20); pdf.text('Experiment Plan Report', 20, 20)
  pdf.setFontSize(12); pdf.text(`Hypothesis: ${experiment.hypothesis}`, 20, 35)
  autoTable(pdf, { head: [['Step','Description','Duration']], body:
    experiment.plan.protocol.map(s => [s.step_number, s.title, `${s.duration_hours}h`]),
    startY: 50 })
  pdf.save('experiment_plan.pdf')
}
Feature F-12  Grant Methods Generator
A dedicated 'Grant Language' tab that uses GPT-4o to rewrite the protocol in formal NIH / NSF / ERC Methods section language. The output is structured exactly as required by each funding body's application guidelines: past tense, third person, with statistical methods described in the format required by study section reviewers.
# app/graph/nodes/grant_methods_generator.py
GRANT_PROMPT = '''
You are an experienced grant writer with a track record of NIH R01 and NSF awards.
Rewrite the following experiment plan as a formal Methods section suitable for submission
to {grant_body} ({grant_body_full}).
Requirements:
  - Past tense, third person (e.g. 'Cells were seeded at 5x10^4 per well...')
  - All reagents must include source and catalog number on first mention
  - Statistical analysis paragraph must appear at the end, specifying:
    test used, alpha level, power, software (e.g. GraphPad Prism 10)
  - Include a sentence about institutional approval (e.g. IACUC) if applicable
  - Target length: 400-600 words
  - Use passive voice throughout
Output only the Methods section text — no preamble, no headers.
'''
async def generate_grant_methods(plan: dict, grant_body: str = 'NIH') -> str:
    grant_map = {'NIH':'National Institutes of Health R01',
                 'NSF':'National Science Foundation',
                 'ERC':'European Research Council'}
    prompt = GRANT_PROMPT.format(grant_body=grant_body,
                                 grant_body_full=grant_map.get(grant_body,''))
    r = await openai_client.chat.completions.create(
        model='gpt-4o', temperature=0.3, max_tokens=1000,
        messages=[{'role':'system','content':prompt},
                  {'role':'user','content':f'Plan: {json.dumps(plan)}'}])
    return r.choices[0].message.content

Multiple scientists can open the same experiment plan simultaneously and see each other's cursors, annotations, and review submissions in real-time. Architecture
# Supabase Realtime uses Presence + Broadcast + Postgres Changes
# 1. PRESENCE — track who is viewing the plan
const channel = supabase.channel(`plan-review-${planId}`)
channel.on('presence', { event: 'sync' }, () => {
  const state = channel.presenceState()
  setActiveScienstists(Object.values(state).flat())
})
channel.track({ user_id: currentUser.id, role: currentUser.role, joined_at: new Date().toISOString() })
# 2. BROADCAST — real-time annotation events

  addAnnotation(payload)  // adds floating comment bubble to the relevant section

# 3. POSTGRES CHANGES — when a review is submitted, all viewers see it immediately

  event: 'INSERT', schema: 'public', table: 'scientist_reviews',
  filter: `plan_id=eq.${planId}`
}, payload => {
  addReviewCard(payload.new)
})
channel.subscribe()

Diff Algorithm
Plan diffs are computed client-side using the fast-diff library (MIT, ~5KB):
import diff from 'fast-diff'  // MIT license
function diffPlanSection(oldText: string, newText: string): DiffChunk[] {
  return diff(oldText, newText).map(([type, value]) => ({
    type: type === 1 ? 'added' : type === -1 ? 'removed' : 'unchanged',
    text: value
  }))
}
